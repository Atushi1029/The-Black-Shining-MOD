DIPLOMACY_GENERATE_WARGOAL_ENABLE_TRIGGER = {
	if = {
		limit = {
			OR = {
				is_ai = yes
				FROM = {
					strength_ratio = {
						tag = ROOT
						ratio > 1.75
					}
				}
				AND = {
					ROOT = { is_major = no }
					FROM = { is_major = yes }
				}
			}
		}
		always = no
	}
	else_if = {
		limit = {
			check_variable = { declare_war_free_tickets < 1 }
			OR = {
				has_socialist_government = yes
				has_dictatorship_government = yes
			}
		}
		command_power > 49
		has_army_experience > 24
		has_navy_experience > 24
		has_air_experience > 24
		has_war_support > 0.74
		num_divisions > 49
	}
	else_if = {
		limit = {
			check_variable = { declare_war_free_tickets < 1 }
			has_democratic_government = yes
		}
		command_power > 49
		has_army_experience > 49
		has_navy_experience > 49
		has_air_experience > 49
		has_war_support > 0.99
		has_war = no
		num_divisions > 74
	}
	else = { always = no }
}
