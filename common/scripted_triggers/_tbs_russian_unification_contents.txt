has_far_eastern_siberia = {
	custom_trigger_tooltip = {
		tooltip = has_far_eastern_siberia_tt
		all_state = { 
			OR = {
				is_controlled_by = ROOT
				NOT = { has_state_flag = far_east_core_states }
			}
		}
	}
}
